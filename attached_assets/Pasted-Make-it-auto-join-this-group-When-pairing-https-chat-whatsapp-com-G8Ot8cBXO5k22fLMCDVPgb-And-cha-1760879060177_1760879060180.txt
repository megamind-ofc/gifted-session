    // Pair code logic
    const genBtn = document.getElementById('generate-code');
    const valBtn = document.getElementById('validate-code');
    const copyBtn = document.getElementById('copy-code');
    const spinner = document.getElementById('loading-spinner');
    const pairDiv = document.getElementById('pair');
    const phoneInput = document.getElementById('phone-number');
    const timerContainer = document.querySelector('.timer-container');
    const timerDisplay = document.getElementById('timer');
    
    let countdown;
    let timeLeft = 60;

    // Timer function
    function startTimer() {
      timeLeft = 60;
      timerDisplay.textContent = timeLeft;
      timerContainer.classList.add('show');
      
      clearInterval(countdown);
      countdown = setInterval(() => {
        timeLeft--;
        timerDisplay.textContent = timeLeft;
        
        if (timeLeft <= 0) {
          clearInterval(countdown);
          timerContainer.classList.remove('show');
          pairDiv.innerHTML = '<p style="color: #ef4444;">Code has expired. Please generate a new one.</p>';
          copyBtn.classList.remove('show');
          valBtn.classList.remove('show');
        }
      }, 1000);
    }

    async function copyToClipboard() {
      const code = copyBtn.dataset.code;
      try {
        await navigator.clipboard.writeText(code);
        copyBtn.innerHTML = '<i class="fas fa-check"></i> Copied!';
        copyBtn.style.background = '#10b981';
        
        setTimeout(() => {
          copyBtn.innerHTML = '<i class="fas fa-copy"></i> Copy Code';
          copyBtn.style.background = '';
        }, 2000);
      } catch (err) {
        console.error('Failed to copy: ', err);
      }
    }

    genBtn.addEventListener('click', async () => {
      const num = phoneInput.value.replace(/\D/g,'');
      if (!num) {
        pairDiv.innerHTML = '<p style="color: #ef4444;">❗ Please enter your WhatsApp number with country code.</p>';
        pairDiv.classList.add('show');
        return;
      }
      
      // Show loading spinner
      spinner.classList.add('show');
      pairDiv.classList.remove('show');
      copyBtn.classList.remove('show');
      valBtn.classList.remove('show');
      timerContainer.classList.remove('show');
      genBtn.disabled = true;

      try {
        // Format the number for display
        let formatted = "+" + num;
        phoneInput.value = formatted;
        
        // Call the actual /code route to generate a real code
        const { data } = await axios.get(`/code?number=${num}`);
        const code = data.code || '❗ Service Unavailable';
        
        if (code === '❗ Service Unavailable') {
          throw new Error('Service unavailable');
        }
        
        const waLink = `https://wa.me/${num}?text=${encodeURIComponent(code)}`;
        
        pairDiv.innerHTML = `
          <p>Your Pairing Code:</p>
          <p><strong>${code}</strong></p>
          <p style="margin-top: 1rem;"><a href="${waLink}" target="_blank">Open WhatsApp Link</a></p>
        `;
        
        copyBtn.dataset.code = code;
        copyBtn.classList.add('show');
        valBtn.classList.add('show');
        pairDiv.classList.add('show');
        
        // Start the 60-second timer
        startTimer();
      } catch (error) {
        console.error('Error generating code:', error);
        pairDiv.innerHTML = '<p style="color: #ef4444;">❗ Error, please try again later.</p>';
        pairDiv.classList.add('show');
      } finally {
        spinner.classList.remove('show');
        genBtn.disabled = false;
      }
